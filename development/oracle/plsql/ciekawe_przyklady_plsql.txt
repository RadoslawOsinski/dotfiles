DECLARE
  l_counter NUMBER;
BEGIN
  SELECT COUNT(1)
  INTO l_counter
  FROM user_tables
  WHERE table_name = 'ABC_TABLE';
  IF l_counter = 0 THEN
	--EXECUTE IMEDIATE DLA DDL. ZBÄ˜DNE PRZY PROSTYCH MODYFIKACJACH DANYCH BEZ ZMIAN SCHEMATU - DML.
    EXECUTE IMMEDIATE ' CREATE TABLE ABC_TABLE
      (ID NUMBER(10,0) NOT NULL PRIMARY KEY,
      PARAM VARCHAR2(11),
      )';
    EXECUTE IMMEDIATE 'CREATE INDEX ABC_TABLE_ID_IDX ON ABC_TABLE(ID)';
  END IF;
END;
